1. Idea de práctica
   Como tfg tenemos Vincen y yo vamos a hacer un videojuego tipo rouguelite el cual 
   tiene mucho que ver con la programación (el propio ataque se construye con comandos)
   La idea es hacer una base de datos tipo wiki para nuestro juego, con objetos, ubicaciones,
   personajes...
2. Organización de BD
   ubicaciones
      id_ubicacion INTEGER PRIMARY KEY
      nombre TEXT NOT NULL
      descripcion TEXT
      tipo TEXT NULL -- (ej. ciudad, mazmorra, mundo)

   personajes
      id_personaje INTEGER PRIMARY KEY
      nombre TEXT NOT NULL
      descripcion TEXT
      id_ubicacion INTEGER REFERENCES ubicaciones(id_ubicacion) NULL
      da_objeto BOOLEAN NOT NULL DEFAULT 0 (false)
      rol TEXT NULL -- (NPC, amigable, mercader, etc.)

   enemigos
      id_enemigo INTEGER PRIMARY KEY
      nombre TEXT NOT NULL
      descripcion TEXT
      nivel INTEGER DEFAULT 1
      id_ubicacion INTEGER REFERENCES ubicaciones(id_ubicacion) NOT NULL
      drop_objetos BOOLEAN NOT NULL DEFAULT 0

   objetos
      id_objeto INTEGER PRIMARY KEY
      nombre TEXT NOT NULL
      descripcion TEXT
      rareza TEXT DEFAULT 'comun' -- (comun, raro, épico...)
      id_ubicacion INTEGER REFERENCES ubicaciones(id_ubicacion) NULL
      id_personaje_dropea INTEGER REFERENCES personajes(id_personaje) NULL
      id_enemigo_dropea INTEGER REFERENCES enemigos(id_enemigo) NULL

   personaje_objeto 
      id_personaje INTEGER REFERENCES personajes(id_personaje) NOT NULL
      id_objeto INTEGER REFERENCES objetos(id_objeto) NOT NULL
      cantidad INTEGER DEFAULT 1
      PRIMARY KEY (id_personaje, id_objeto)


   enemigo_objeto (opcional: relación muchos-a-muchos para dropeos múltiples)
      id_enemigo INTEGER REFERENCES enemigos(id_enemigo) NOT NULL
      id_objeto INTEGER REFERENCES objetos(id_objeto) NOT NULL
      probabilidad FLOAT DEFAULT 1.0 -- probabilidad de drop
      PRIMARY KEY (id_enemigo, id_objeto)
